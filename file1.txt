package com.candidjava.code;
 
class OddNumber {
	public static void main(String args[])
	 {
		System.out.println("The Odd Numbers are:");
		for (int i = 1; i <= 100; i++) 
	{
			if (i % 2 != 0) {
				System.out.print(i + " ");
			}
		for (int j = 1; j <= 100; j++) 
	{
			if (j % 2 != 0) {
				System.out.print(j + " ");
			}
		}
	}
}
